Enter:

            code:

             var data = [
                        {
                            x:10,
                            y:10,
                            width:5,
                            height:40
                        },{
                            x:40,
                            y:10,
                            width:100,
                            height:40
                        }];

            var svg2 = d3.select("#d3_show2")
                        .append("svg")
                        .attr("width", 200)
                        .attr("height", 200);

            svg2.selectAll('rect').data(data).enter()
                    .append('rect')
                    .attr('x', function(d){ return d.x})
                    .attr('y', function(d){ return d.y})
                    .attr("width", function(d){ return d.width})
                    .attr("height", function(d){ return d.height});
            
                <div class="myborder" id="d3_show2"></div>
